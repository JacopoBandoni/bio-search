 68def new_class(name, bases=(), kwds=None, exec_body=None):
 69    """Create a class object dynamically using the appropriate metaclass."""
 70    resolved_bases = resolve_bases(bases)
 71    meta, ns, kwds = prepare_class(name, resolved_bases, kwds)
 72    if exec_body is not None:
 73        exec_body(ns)
 74    if resolved_bases is not bases:
 75        ns['__orig_bases__'] = bases
 76    return meta(name, resolved_bases, ns, **kwds)

 78def resolve_bases(bases):
 79    """Resolve MRO entries dynamically as specified by PEP 560."""
 80    new_bases = list(bases)
 81    updated = False
 82    shift = 0
 83    for i, base in enumerate(bases):
 84        if isinstance(base, type) and not isinstance(base, GenericAlias):
 85            continue
 86        if not hasattr(base, "__mro_entries__"):
 87            continue
 88        new_base = base.__mro_entries__(bases)
 89        updated = True
 90        if not isinstance(new_base, tuple):
 91            raise TypeError("__mro_entries__ must return a tuple")
 92        else:
 93            new_bases[i+shift:i+shift+1] = new_base
 94            shift += len(new_base) - 1
 95    if not updated:
 96        return bases
 97    return tuple(new_bases)

 99def prepare_class(name, bases=(), kwds=None):
100    """Call the __prepare__ method of the appropriate metaclass.
101
102    Returns (metaclass, namespace, kwds) as a 3-tuple
103
104    *metaclass* is the appropriate metaclass
105    *namespace* is the prepared class namespace
106    *kwds* is an updated copy of the passed in kwds argument with any
107    'metaclass' entry removed. If no kwds argument is passed in, this will
108    be an empty dict.
109    """
110    if kwds is None:
111        kwds = {}
112    else:
113        kwds = dict(kwds) # Don't alter the provided mapping
114    if 'metaclass' in kwds:
115        meta = kwds.pop('metaclass')
116    else:
117        if bases:
118            meta = type(bases[0])
119        else:
120            meta = type
121    if isinstance(meta, type):
122        # when meta is a type, we first determine the most-derived metaclass
123        # instead of invoking the initial candidate directly
124        meta = _calculate_meta(meta, bases)
125    if hasattr(meta, '__prepare__'):
126        ns = meta.__prepare__(name, bases, **kwds)
127    else:
128        ns = {}
129    return meta, ns, kwds
130
131def _calculate_meta(meta, bases):
132    """Calculate the most derived metaclass."""
133    winner = meta
134    for base in bases:
135        base_meta = type(base)
136        if issubclass(winner, base_meta):
137            continue
138        if issubclass(base_meta, winner):
139            winner = base_meta
140            continue
141        # else:
142        raise TypeError("metaclass conflict: "
143                        "the metaclass of a derived class "
144                        "must be a (non-strict) subclass "
145                        "of the metaclasses of all its bases")
146    return winner

148class DynamicClassAttribute:
149    """Route attribute access on a class to __getattr__.
150
151    This is a descriptor, used to define attributes that act differently when
152    accessed through an instance and through a class.  Instance access remains
153    normal, but access to an attribute through a class will be routed to the
154    class's __getattr__ method; this is done by raising AttributeError.
155
156    This allows one to have properties active on an instance, and have virtual
157    attributes on the class with the same name (see Enum for an example).
158
159    """
160    def __init__(self, fget=None, fset=None, fdel=None, doc=None):
161        self.fget = fget
162        self.fset = fset
163        self.fdel = fdel
164        # next two lines make DynamicClassAttribute act the same as property
165        self.__doc__ = doc or fget.__doc__
166        self.overwrite_doc = doc is None
167        # support for abstract methods
168        self.__isabstractmethod__ = bool(getattr(fget, '__isabstractmethod__', False))
169
170    def __get__(self, instance, ownerclass=None):
171        if instance is None:
172            if self.__isabstractmethod__:
173                return self
174            raise AttributeError()
175        elif self.fget is None:
176            raise AttributeError("unreadable attribute")
177        return self.fget(instance)
178
179    def __set__(self, instance, value):
180        if self.fset is None:
181            raise AttributeError("can't set attribute")
182        self.fset(instance, value)
183
184    def __delete__(self, instance):
185        if self.fdel is None:
186            raise AttributeError("can't delete attribute")
187        self.fdel(instance)
188
189    def getter(self, fget):
190        fdoc = fget.__doc__ if self.overwrite_doc else None
191        result = type(self)(fget, self.fset, self.fdel, fdoc or self.__doc__)
192        result.overwrite_doc = self.overwrite_doc
193        return result
194
195    def setter(self, fset):
196        result = type(self)(self.fget, fset, self.fdel, self.__doc__)
197        result.overwrite_doc = self.overwrite_doc
198        return result
199
200    def deleter(self, fdel):
201        result = type(self)(self.fget, self.fset, fdel, self.__doc__)
202        result.overwrite_doc = self.overwrite_doc
203        return result

243def coroutine(func):
244    """Convert regular generator function to a coroutine."""
245
246    if not callable(func):
247        raise TypeError('types.coroutine() expects a callable')
248
249    if (func.__class__ is FunctionType and
250        getattr(func, '__code__', None).__class__ is CodeType):
251
252        co_flags = func.__code__.co_flags
253
254        # Check if 'func' is a coroutine function.
255        # (0x180 == CO_COROUTINE | CO_ITERABLE_COROUTINE)
256        if co_flags & 0x180:
257            return func
258
259        # Check if 'func' is a generator function.
260        # (0x20 == CO_GENERATOR)
261        if co_flags & 0x20:
262            # TODO: Implement this in C.
263            co = func.__code__
264            # 0x100 == CO_ITERABLE_COROUTINE
265            func.__code__ = co.replace(co_flags=co.co_flags | 0x100)
266            return func
267
268    # The following code is primarily to support functions that
269    # return generator-like objects (for instance generators
270    # compiled with Cython).
271
272    # Delay functools and _collections_abc import for speeding up types import.
273    import functools
274    import _collections_abc
275    @functools.wraps(func)
276    def wrapped(*args, **kwargs):
277        coro = func(*args, **kwargs)
278        if (coro.__class__ is CoroutineType or
279            coro.__class__ is GeneratorType and coro.gi_code.co_flags & 0x100):
280            # 'coro' is a native coroutine object or an iterable coroutine
281            return coro
282        if (isinstance(coro, _collections_abc.Generator) and
283            not isinstance(coro, _collections_abc.Coroutine)):
284            # 'coro' is either a pure Python generator iterator, or it
285            # implements collections.abc.Generator (and does not implement
286            # collections.abc.Coroutine).
287            return _GeneratorWrapper(coro)
288        # 'coro' is either an instance of collections.abc.Coroutine or
289        # some other object -- pass it through.
290        return coro
291
292    return wrapped

245    """Convert regular generator function to a coroutine."""
246
247    if not callable(func):
248        raise TypeError('types.coroutine() expects a callable')
249
250    if (func.__class__ is FunctionType and
251        getattr(func, '__code__', None).__class__ is CodeType):
252
253        co_flags = func.__code__.co_flags
254
255        # Check if 'func' is a coroutine function.
256        # (0x180 == CO_COROUTINE | CO_ITERABLE_COROUTINE)
257        if co_flags & 0x180:
258            return func
259
260        # Check if 'func' is a generator function.
261        # (0x20 == CO_GENERATOR)
262        if co_flags & 0x20:
263            # TODO: Implement this in C.
264            co = func.__code__
265            # 0x100 == CO_ITERABLE_COROUTINE
266            func.__code__ = co.replace(co_flags=co.co_flags | 0x100)
267            return func
268
269    # The following code is primarily to support functions that
270    # return generator-like objects (for instance generators
271    # compiled with Cython).
272
273    # Delay functools and _collections_abc import for speeding up types import.
274    import functools
275    import _collections_abc
276    @functools.wraps(func)
277    def wrapped(*args, **kwargs):
278        coro = func(*args, **kwargs)
279        if (coro.__class__ is CoroutineType or
280            coro.__class__ is GeneratorType and coro.gi_code.co_flags & 0x100):
281            # 'coro' is a native coroutine object or an iterable coroutine
282            return coro
283        if (isinstance(coro, _collections_abc.Generator) and
284            not isinstance(coro, _collections_abc.Coroutine)):
285            # 'coro' is either a pure Python generator iterator, or it
286            # implements collections.abc.Generator (and does not implement
287            # collections.abc.Coroutine).
288            return _GeneratorWrapper(coro)
289        # 'coro' is either an instance of collections.abc.Coroutine or
290        # some other object -- pass it through.
291        return coro
292
293    return wrapped
